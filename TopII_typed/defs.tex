\node (d) [right of=c] { $ #8 $ };

\draw[-to] (a) to node { $ #2 $ } (c);
\draw[-to] (b) to node { $ #3 $ } (d);
\draw[-to] (c) to node { $ #4 $ } (d);
\end{tikzpicture}
}
